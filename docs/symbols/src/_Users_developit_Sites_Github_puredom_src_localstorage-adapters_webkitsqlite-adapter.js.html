1 /**	@class Storage adapter that persists data into HTML5 LocalStorage.
  2  *	@name puredom.LocalStorage.adapters.WebkitSQLite
  3  */
  4 puredom.LocalStorage.addAdapter('WebkitSQLite', /** @lends puredom.LocalStorage.adapters.WebkitSQLite */ {
  5 	
  6 	/** The default cookie ID to use for database storage */
  7 	defaultName : 'db',
  8 	
  9 	
 10 	/**	Default database identification info. */
 11 	dbInfo : {
 12 		name : 'PuredomLocalStorage',
 13 		table : 'storage',
 14 		version : '1.0',
 15 		displayName : 'Cache, Settings and Storage',
 16 		quota : 200000,					// 200 k
 17 		minimumQuota : 10000			// 10 k
 18 	},
 19 	
 20 	
 21 	/**	This adapter is the fastest storage mechanism for Webkit. <br />
 22 	 *	Web SQL has also be discontinued in favour of IndexedDB. Who knew SQL would turn out to be annoying? ... <br />
 23 	 *	In terms of complexity, clearly LocalStorage is better, but for now this adapter can stay at the top of the list.
 24 	 */
 25 	rating : 80,
 26

 28 	/** Test if this adapter will work in the current environment */
 29 	test : function(storage) {
 30 		return !!window.openDatabase;
 31 	},

 34 	/** Load the persisted DB */
 35 	load : function(storage, callback) {
 36 		var db = this._getDatabase(storage),
 37 			key = this._getKey(storage),
 38 			table = this.dbInfo.table,
 39 			errorCB;
 40 		
 41 		callback = callback || this._nullCallback;
 42 		
 43 		if (db) {
 44 			errorCB = function(error) {
 45 				if (error.message.indexOf('no such table')>-1) {
 46 					callback();
 47 				}
 48 				else {
 49 					puredom.log('WebkitSQLite Adapter Error (load): ' + error.message);
 50 				}
 51 			};
 52 			
 53 			db.transaction(function(tx) {
 54 				if (tx) {
 55 					tx.executeSql('SELECT key,value FROM '+table+' WHERE key=?', [key], function(tx, result) {
 56 						var rows = [];
 57 						if (result && result.rows) {
 58 							for (var x=0; x<result.rows.length; x++) {
 59 								rows.push(result.rows.item(x));
 60 							}
 61 						}
 62 						if (rows.length>0) {
 63 							callback(puredom.json.parse(rows[0].value));
 64 						}
 65 						else {
 66 							callback();
 67 						}
 68 					});
 69 				}
 70 				else {
 71 					callback(false);
 72 				}
 73 			}, errorCB);
 74 		}
 75 	},
 76 	
 77 	
 78 	/** Save the DB to persistence. */
 79 	save : function(storage, data, callback) {
 80 		var key = this._getKey(storage),
 81 			table = this.dbInfo.table;
 82 		callback = callback || this._nullCallback;
 83 		
 84 		this._requireDatabase(storage, function(db) {
 85 			var errorCB,
 86 				jobs = 1,
 87 				jobComplete;
 88 			if (db) {
 89 				errorCB = function(error) {
 90 					puredom.log('WebkitSQLite Adapter Error (save): ' + error.message);
 91 				};
 92 				
 93 				jobComplete = function() {
 94 					jobs -= 1;
 95 					if (jobs<=0) {
 96 						if (callback) {
 97 							callback(true);
 98 						}
 99 						callback = storage = data = errorCB = jobComplete = db = null;
100 					}
101 				};
102 				
103 				data = puredom.json.stringify(data);
104 				
105 				db.transaction(function(tx){
106 					tx.executeSql('INSERT OR REPLACE INTO '+table+' (key,value) VALUES(?,?)', [key,data], jobComplete);
107 				}, errorCB);
108 			}
109 			else {
110 				callback(false);
111 				callback = null;
112 			}
113 		});
114 	},
115 	
116 	
117 	/**	@private */
118 	_getDatabase : function(storage) {
119 		var quota = this.dbInfo.quota,
120 			db;
121 		if (this._currentDb) {
122 			return this._currentDb;
123 		}
124 		while (!db && quota>this.dbInfo.minimumQuota) {
125 			try {
126 				db = openDatabase(this.dbInfo.name, this.dbInfo.version, this.dbInfo.displayName, quota);
127 			}catch(err){}
128 			if (!db) {
129 				quota /= 10;
130 			}
131 		}
132 		if (db) {
133 			this._currentDb = db;
134 		}
135 		else {
136 			puredom.log('LocalStorage ERROR: WebkitSQLite Adapter failed to open database.');
137 		}
138 		return db || false;
139 	},
140 	
141 	
142 	/**	@private */
143 	_requireDatabase : function(storage, callback) {
144 		var self = this,
145 			db = this._getDatabase(storage),
146 			table = this.dbInfo.table;
147 		callback = callback || this._nullCallback;
148 		
149 		if (db) {
150 			self._createTable(db, function() {
151 				callback(db);
152 				self = callback = storage = db = null;
153 			});
154 		}
155 		else {
156 			callback(db, false);
157 			self = callback = storage = null;
158 		}
159 	},
160 	
161 	
162 	/**	@private */
163 	_createTable : function(db, callback) {
164 		var table = this.dbInfo.table;
165 		callback = callback || this._nullCallback;
166 		if (db) {
167 			db.transaction(function(tx) {
168 				tx.executeSql('CREATE TABLE IF NOT EXISTS '+table+' (key TEXT UNIQUE, value TEXT)', [], function(tx, result) {
169 					callback(true);
170 					callback = db = null;
171 				}, function(tx, error) {
172 					callback(false);
173 				});
174 			});
175 		}
176 		else {
177 			callback(false);
178 		}
179 	},
180 	
181 	
182 	/** Get the key for a storage object
183 	 *	@private
184 	 */
185 	_getKey : function(storage) {
186 		return (storage.id || this.defaultName || '') + '';
187 	},
188 	
189 	
190 	/** @private */
191 	_nullCallback : function(){}
192 	
193 });
