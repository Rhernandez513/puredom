1 /**	@class Storage adapter that persists data into HTML5 LocalStorage.
  2  *	@name puredom.LocalStorage.adapters.LocalStorage
  3  */
  4 puredom.LocalStorage.addAdapter('LocalStorage', /** @lends puredom.LocalStorage.adapters.LocalStorage */ {
  5 	
  6 	/**	The default root key ID to use for accessing localStorage */
  7 	defaultName : 'db',
  8 	
  9 	
 10 	/**	This adapter is a very good storage mechanism, so its rating is 60. */
 11 	rating : 60,
 12 	
 13 	
 14 	/**	Test if this adapter will work in the current environment */
 15 	test : function(storage) {
 16 		var available = ('localStorage' in window) && typeof window.localStorage.hasOwnProperty==='function',
 17 			prev,
 18 			val = puredom.json({a:'a',b:4/3,c:true,d:null});
 19 		if (available) {
 20 			try {
 21 				prev = localStorage.__test;
 22 				localStorage.__test = val;
 23 				if (localStorage.__test!==val) {
 24 					available = false;
 25 				}
 26 				localStorage.__test = prev;
 27 				if (prev===undefined) {
 28 					delete localStorage.__test;
 29 				}
 30 			} catch(err) {
 31 				available = false;
 32 			}
 33 		}
 34 		return available;
 35 	},
 36 	
 37 	
 38 	/**	Load the persisted DB */
 39 	load : function(storage, callback) {
 40 		var key = this._getKey(storage),
 41 			data;
 42 		if (localStorage.hasOwnProperty(key)) {
 43 			data = puredom.json.parse(localStorage[key]);
 44 		}
 45 		if (callback) {
 46 			callback(data);
 47 		}
 48 		return data;
 49 	},
 50 	
 51 	/**	Save the DB to localStorage */
 52 	save : function(storage, data, callback) {
 53 		var key = this._getKey(storage);
 54 		if (data===undefined) {
 55 			delete localStorage[key];
 56 		}
 57 		else {
 58 			localStorage[key] = puredom.json.stringify(data);
 59 		}
 60 		if (callback) {
 61 			callback(true);
 62 		}
 63 		return true;
 64 	},
 65 	
 66 	
 67 	/**	Get the key for a storage object
 68 	 *	@private
 69 	 */
 70 	_getKey : function(storage) {
 71 		return (storage.id || this.defaultName || '') + '';
 72 	}
 73 	
 74 });
