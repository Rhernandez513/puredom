 55 	/**	Make multiple HTTP requests in order, firing the callback only when all have completed.
 56 	 *	<br /><b>Callback Format:</b><br />
 57 	 *	@example
 58 	 *	callback(
 59 	 *		success   // {Boolean} - did *any* requests succeed?
 60 	 *		responses // {Array}   - responses corresponding to the provided resources.
 61 	 *		successes // {Number}  - how many requests succeeded (status<400)
 62 	 *		failures  // {Number}  - how many requests failed (status>=400)
 63 	 *	);
 64 	 *	@param {Object[]} resources		An array of resource objects, with format as described in {@link puredom.net.request} options.
 65 	 *	@param {Function} [callback]	A function to call once all requests have completed, with signature <code>function(success, responses, successes, failures)</code>. [See description]
 66 	 *	@returns {Boolean} returns false if no resources were provided.
 67 	 */
 68 	multiLoad : function(resources, callback) {
 69 		if (!resources) {
 70 			return false;
 71 		}
 72 		var cur = -1,
 73 			max = resources.length,
 74 			allData = [],
 75 			trues = 0,
 76 			falses = 0,
 77 			loaded, loadNext;
 78 		
 79 		/** @inner */
 80 		loaded = function(result, data) {
 81 			if (result && data) {
 82 				var res = resources[cur],
 83 					d = data;
 84 				if (res.process && res.process.call) {
 85 					d = res.process(d);
 86 					if (d===undefined) {
 87 						d = data;
 88 					}
 89 				}
 90 				allData.push(d);
 91 				if (callback) {
 92 					callback(trues>0, allData, trues, falses);
 93 				}
 94 				loaded = loadNext = resources = allData = callback = null;
 95 			}
 96 			else {
 97 				loadNext();
 98 			}
 99 		};
100 		
101 		/** @inner */
102 		loadNext = function() {
103 			cur += 1;
104 			var res = resources[cur],
105 				d = typeof(res)==='string' ? {url:res} : res;
106 			if (d) {
107 				http.request(d, loaded);
108 			}
109 			else {
110 				if (cur<max) {
111 					loadNext();
112 				}
113 				else {
114 					callback(false, null, null, "No resources were available.");
115 				}
116 			}
117 		};
118 		
119 		loadNext();
120 		
121 		return true;
122 	},

125 	/**	Construct and send an HTTP request based on a configuration object (options). <br />
126 	 *	<strong>Options:</strong> <br />
127 	 *		<table class="options"><tbody>
128 	 *		<tr><td>{String}</td><td><b>url</b></td><td>Required. A URL to which the request should be sent.</td></tr>
129 	 *		<tr><td>{String}</td><td><b>type</b></td><td>An explicit request type (HTTP verb). Generally "GET" or "POST". Overrides other methods of setting request type.</td></tr>
130 	 *		<tr><td>{Object|String}</td><td><b>post</b></td><td>Post body. {Object}s are converted to a form-encoded body using {@link puredom.parameterize}. {String}s are used as the POST body with no manipulation. If this value is set, the request type will be POST unless overridden by *type*.</td></tr>
131 	 *		<tr><td>{Function}</td><td><b>callback</b></td><td>A function to be called when the request has completed, with signature: <code>function({Boolean} success, {Object|String|Document} data)</code></td></tr>
132 	 *		<tr><td>{Object}</td><td><b>headers</b></td><td>Hashmap of request headers. (key-value)</td></tr>
133 	 *		<tr><td>{String}</td><td><b>contentTypeOverride</b></td><td>If set, overrides the *Content-Type* header returned by the server.</td></tr>
134 	 *		</tbody></table>
135 	 *	@param {Object} options			Define request options
136 	 *	@param {Function} [callback]	A callback function, used if options.callback is not set.
137 	 *	@returns {puredom.net.HttpRequest} An HTTP request object
138 	 */
139 	request : function(options) {
140 		var opt, self;
141 		if (!options.url) {
142 			return false;
143 		}
144 		self = this;
145 		options = options || {};
146 		opt = new puredom.net.HttpRequest({
147 			url			: options.url,
148 			type		: options.type || (options.post ? "POST" : "GET"),
149 			callback	: options.callback || arguments[1] || function(){},
150 			post		: options.post,
151 			headers		: options.headers
152 		});
153 		if (options.contentTypeOverride) {
154 			opt.contentTypeOverride = options.contentTypeOverride;
155 			delete options.contentTypeOverride;
156 		}
157 		
158 		this.createXHR(opt.url, function(xhrCarrier) {
159 			opt.request = xhrCarrier.xhr;
160 			opt._xdrFrame = xhrCarrier.frame;
161 			xhrCarrier = null;
162 			
163 			if (opt.post && puredom.typeOf(opt.post)==='object') {
164 				opt.post = puredom.parameterize(opt.post);
165 				if (opt.post.substring(0,1)==='?') {
166 					opt.post = opt.post.substring(1);
167 				}
168 			}
169 			/** @private */
170 			opt.request.onreadystatechange = function() {
171 				var contentType, data, i;
172 				
173 				// for proxied XHR only:
174 				if (opt.request._orig) {
175 					opt.request.readyState = opt.request._orig.readyState;
176 					opt.request.status = opt.request._orig.status;
177 					opt.request.responseText = opt.request._orig.responseText;
178 					opt.request.responseXML = opt.request._orig.responseXML;
179 				}
180 				
181 				if (opt.request.readyState===4) {
182 					// The cross-domain frame can now be re-used.
183 					if (opt._xdrFrame) {
184 						setTimeout(function() {
185 							self._freeIframes.push(opt._xdrFrame);
186 							self = null;
187 						}, 100);
188 					}
189 					
190 					opt.status = opt.request.status;
191 					if (opt.contentTypeOverride) {
192 						contentType = opt.contentTypeOverride.toLowerCase();
193 					}
194 					else {
195 						try {
196 							contentType = (opt.request.getResponseHeader("Content-Type")).toLowerCase();
197 						} catch(err) {}
198 						contentType = contentType || "";
199 					}
200 					opt.responseText = opt.request.responseText;
201 					
202 					if (contentType.match(/\/(json|javascript)$/gm) || contentType==="json") {
203 						opt.responseType = "json";
204 						data = opt.responseJSON = null;
205 						try {
206 							data = opt.responseJSON = JSON.parse(opt.request.responseText.replace(/^[^\[\{]*(.*)[^\[\{]*$/g,'$1'));
207 						}catch(jsonParseError){
208 							opt.jsonParseError = true;
209 						}
210 					}
211 					else if (contentType==="application/xml" || contentType==="xml") {
212 						opt.responseType = "xml";
213 						data = opt.responseXML = opt.request.responseXML;
214 					}
215 					else {
216 						opt.responseType = "text";
217 						data = opt.responseText;
218 					}
219 					
220 					if (opt.callback) {
221 						opt.callback(opt.request.status<400, data);
222 					}
223 				}
224 			};
225 			opt.request.open(opt.type, opt.url, opt.async!==false);
226 			opt.request.setRequestHeader('content-type', 'application/x-www-form-urlencoded');
227 			opt.request.setRequestHeader('x-requested-with', 'XMLHttpRequest');
228 			if (opt.headers) {
229 				for (var h in opt.headers) {
230 					if (opt.headers.hasOwnProperty(h)) {
231 						try {
232 							opt.request.setRequestHeader(h, opt.headers[h]);
233 						} catch(err) {}
234 					}
235 				}
236 			}
237 			opt.request.send(opt.post || null);
238 		});
239 		return opt;
240 	},

243 	/**	When called as a function, <code>puredom.net.jsonp()</code> is an alias of {@link puredom.net.jsonp.get}.
244 	 *	@namespace JSONp Implementation. JSONp is GET-only and works across domains, but the server must support the JSONp pattern.
245 	 *	@function
246 	 *	@returns {puredom.net.jsonp.Request} jsonpRequest
247 	 */
248 	jsonp : (function() {
249 		/** @namespace JSONp-related functionality.
250 		 *	@name puredom.net.jsonp
251 		 *	@private
252 		 */
253 		var jsonp = function() {
254 				return jsonp.get.apply(jsonp, arguments);
255 			},
256 			reqIndex = 0;
257 		
258 		/**	Initiate a JSONp request.
259 		 *	@name puredom.net.jsonp.get
260 		 *	@function
261 		 *	@param {String} url			The service URL, including querystring parameters.
262 		 *	@param {Object} [options]		A hash of available options.
263 		 *	@param {String} [options.url=url]		The service URL
264 		 *	@param {Object} [options.params]		GET parameters as an object.
265 		 *	@param {Function} [options.callback]	A function to handle the data once received.
266 		 *	@param {Number} [options.timeout=10]	A number of seconds to wait before triggering failure.
267 		 *	@param {Function} callback	A function that gets called when the request returns.
268 		 *	@returns {puredom.net.jsonp.Request} jsonpRequest
269 		 */
270 		jsonp.get = function(url, options, callback) {
271 			var script, requestObj, callbackId, tmp;
272 			
273 			if (puredom.typeOf(options)==='function') {
274 				if (callback && puredom.typeOf(callback)==='object') {
275 					tmp = callback;
276 				}
277 				callback = options;
278 				if (tmp) {
279 					options = tmp;
280 				}
281 			}
282 			options = options || {};
283 			if (options.callback && !callback) {
284 				callback = options.callback;
285 			}
286 			if (!options.timeout) {
287 				options.timeout = 10;
288 			}
289 			url = url || options.url;
290 			
291 			if (!url) {
292 				return false;
293 			}
294 			
295 			if (options.params && puredom.parameterize) {
296 				url += (url.indexOf('?')>-1?'&':'?') + puredom.querystring.stringify(options.params);
297 			}
298 			
299 			reqIndex += 1;
300 			
301 			options.callback = callbackId = "puredom_net_jsonp_"+reqIndex;
302 			(function(jsonp, reqIndex) {
303 				/**	@ignore */
304 				window[options.callback] = function(data) {
305 					var e;
306 					if (callback) {
307 						try {
308 							callback(data);
309 						} catch(err) {
310 							e = err;
311 						}
312 						callback = null;
313 					}
314 					if (requestObj) {
315 						requestObj.stop();
316 						requestObj = null;
317 					}
318 					if (e) {
319 						throw(e);
320 					}
321 				};
322 			}());
323 			
324 			if (url.indexOf('{!callback}')>-1) {
325 				url = url.replace('{!callback}', callbackId);
326 			}
327 			else {
328 				url += (url.indexOf('?')<0?'?':'&') + encodeURIComponent(options.callbackParam || 'callback') + '=' + encodeURIComponent(callbackId);
329 			}
330 			
331 			if (!this._head) {
332 				tmp = document.getElementsByTagName('head');
333 				this._head = tmp && tmp[0];
334 			}
335 			
336 			script = puredom.el({
337 				type : 'script',
338 				attributes : {
339 					src		: url,
340 					async	: 'async',
341 					type	: 'text/javascript'
342 				},
343 				parent : this._head || document.body
344 			});
345 			
346 			/**	@class Represents a JSONp request.
347 			 *	@name puredom.net.jsonp.Request
348 			 */
349 			requestObj = /** @lends puredom.net.jsonp.Request# */ {
350 				/**	The request's callback ID */
351 				id : callbackId,
352 				
353 				/**	Attempt to stop the request. */
354 				stop : function() {
355 					if (requestObj._timer) {
356 						clearTimeout(requestObj._timer);
357 					}
358 					window[callbackId] = null;
359 					try {
360 						delete window[callbackId];
361 					}catch(err){}
362 					callback = null;
363 					script.attr('src', 'about:blank').remove();
364 					callbackId = requestObj = script = null;
365 				}
366 			};
367 			
368 			if (options.timeout && options.timeout>0) {
369 				requestObj._timer = setTimeout(function() {
370 					if (callback) {
371 						callback({
372 							_requestTimedOut : true,
373 							_jsonpTimedout : true,
374 							success : false,
375 							result : false
376 						});
377 					}
378 					if (requestObj) {
379 						requestObj.stop();
380 					}
381 				}, Math.round(options.timeout*1000));
382 			}
383 			
384 			url = options = tmp = null;
385 			
386 			return requestObj;
387 		};
388 		
389 		return jsonp;
390 	}()),

401 	/** Asynchronously create an XMLHttpRequest object, automatically instantiating it from within an iframe if the TLD matches the page's TLD.
402 	 *	@private
403 	 */
404 	createXHR : function(url, callback) {
405 		var isCrossDomain = false,
406 			self = this,
407 			domain, frame, xhr, _loadHandler, timer, i;
408 		this._xhrIndex += 1;
409 		if (url) {
410 			domain = (/^[a-z]{3,9}\:\/\/([^\/\?#]+)/gim).exec(url);
411 			domain = domain && domain[1];
412 			if (domain && domain!==location.hostname) {
413 				isCrossDomain = true;
414 				document.domain = location.hostname.match(/[^.]+\.[^.]+$/gim)[0];
415 				for (i=0; i<this._freeIframes.length; i++) {
416 					if (this._freeIframes[i].getAttribute('data-xhr-domain')===domain) {
417 						frame = this._freeIframes.splice(i,1)[0];
418 						break;
419 					}
420 				}
421 				if (frame) {
422 					callback(self._createXHRObj(frame.contentWindow, frame));
423 				}
424 				else {
425 					frame = document.createElement('iframe');
426 					frame.style.cssText = "position:absolute; left:0; top:-1000px; width:1px; height:1px; border:none; overflow:hidden;";
427 					/** @inner */
428 					_loadHandler = function() {
429 						var win, body;
430 						try {
431 							win = frame.contentWindow;
432 							body = win && win.document && win.document.domain===document.domain && win.document.body;
433 						}catch(err){
434 							body = null;
435 						}
436 						if (body && body.innerHTML) {
437 							clearInterval(timer);
438 							frame.onload = frame.onerror = null;
439 							callback(self._createXHRObj(win, frame));
440 							self = callback = xhr = frame = domain = _loadHandler = timer = null;
441 						}
442 					};
443 					frame.onload = frame.onerror = _loadHandler;
444 					frame.setAttribute('src', location.protocol+'//'+domain+'/xd_receiver.html');
445 					frame.setAttribute('role', 'presentation');
446 					frame.setAttribute('tabindex', '-1');
447 					frame.setAttribute('data-xhr-domain', domain);
448 					document.body.appendChild(frame);
449 					timer = setInterval(_loadHandler, 50);
450 				}
451 			}
452 		}
453 		if (!isCrossDomain) {
454 			xhr = this._createXHRObj();
455 			callback(xhr);
456 			return xhr;
457 		}
458 	},

461 	/** @private */
462 	_createXHRObj : function(win, frame) {
463 		var xmlHttp;
464 		win = win || window;
465 		
466 		try {
467 			xmlHttp = new win.XMLHttpRequest();
468 		} catch(err) {
469 			try {
470 				xmlHttp = new win.ActiveXObject("Msxml2.XMLHTTP");
471 			} catch(err2) {
472 				xmlHttp = new win.ActiveXObject("Microsoft.XMLHTTP");
473 			}
474 		}
475 		return {
476 			xhr : xmlHttp,
477 			frame : frame
478 		};
479 	}
