1 /**	Generate a functional JavaScript interface on top of a REST-like API from a JSON API description.
  2  *	@constructor Creates a new NativeAPI instance.
  3  *	@param {Object} api		The API description
  4  *	@returns {Object}	An object with methods corresponding to the API object's method descriptions.
  5  */

 29 	/**	@private */
 30 	getQueryStringFromObj = function (obj) {
 31 		var querystring = "",
 32 			x, i;
 33 		for (x in obj) {
 34 			if (obj[x]!==null && obj[x]!==undefined && obj[x].constructor!==Function && obj[x].constructor!==objConstructor && puredom.typeOf(obj[x])!=='object' && !isArray(obj[x])) {
 35 				querystring += "&" + encodeURIComponent(x) + "=" + encodeURIComponent(obj[x]);
 36 			}
 37 			else if (obj[x] && isArray(obj[x])) {
 38 				for (i=0; i<obj[x]; i++) {
 39 					querystring += "&" + encodeURIComponent(x) + "[]=" + encodeURIComponent(obj[x]);
 40 				}
 41 			}
 42 		}
 43 		return querystring;
 44 	};

 46 	/**	@private */
 47 	shallowObjectCopy = function(base, args) {
 48 		var i, p, obj;
 49 		for (i=1; i<arguments.length; i++) {
 50 			obj = arguments[i];
 51 			for (p in obj) {
 52 				if (obj.hasOwnProperty(p)) {
 53 					base[p] = obj[p];
 54 				}
 55 			}
 56 		}
 57 		return base;
 58 	};

137 	/** @private */
138 	priv.getCacheKey = function(method, options) {
139 		var key = '', list=[], i, props=[];
140 		options = puredom.extend({}, options || {});
141 		delete options.callback;
142 		key = (method.type || '') + '||' + method.endpoint + '||';
143 		for (i in options) {
144 			if (options.hasOwnProperty(i) && i!=='_cache' && i!=='_nocache' && i!=='_cache_deleteonly' && i!=='callback') {
145 				props.push(i);
146 			}
147 		}
148 		props.sort();
149 		for (i=0; i<props.length; i++) {
150 			list.push(encodeURIComponent('o_'+props[i]) + '=' + encodeURIComponent(options[props[i]]+''));
151 		}
152 		key += list.join('&');
153 		return key;
154 	};

193 	/** @private Validate submitted values against an API method definition */
194 	priv.validateParameters = function(options, method) {
195 		var requiredParams, name, p, pType, errorType, inputField, error,
196 			response = {
197 				errors : [],
198 				message : ''
199 			},
200 			baseErrorMessage = {
201 				nativeApiError : true,
202 				clientSideError : true
203 			};
204 		
205 		if (method.parameters) {
206 			requiredParams = {};
207 			for (name in method.parameters) {
208 				if (method.parameters.hasOwnProperty(name)) {
209 					p = method.parameters[name];
210 					pType = puredom.typeOf(p);
211 					// convert direct references to constructors to their lowercase'd names:
212 					if (pType==='function' && p.name) {
213 						pType = 'string';
214 						p = p.name.toLowerCase();
215 					}
216 					else if (pType==='regexp' || p.constructor===RegExp) {
217 						pType = 'string';
218 						p = '/' + p.source + '/' + (p.global?'g':'') + (p.ignoreCase?'i':'') + (p.multiline?'m':'');
219 					}
220 					// all validation types are strings
221 					if (pType==='string') {
222 						if (p.substring(0,1)==='/') {
223 							p = (/^\/(.*?)\/([gim]*?)$/gim).exec(p);
224 							requiredParams[name] = {
225 								validate : 'regex',
226 								against : new RegExp(p[1] || '', p[2] || '')
227 							};
228 						}
229 						else {
230 							requiredParams[name] = {
231 								validate : 'type',
232 								against : p.toLowerCase()
233 							};
234 						}
235 					}
236 				}
237 			}
238 		}
239 		
240 		if (requiredParams) {
241 			for (name in requiredParams) {
242 				if (requiredParams.hasOwnProperty(name)) {
243 					pType = requiredParams[name].validate;
244 					p = requiredParams[name].against;
245 					inputField = options[name];
246 					error = null;
247 					if (pType==='regex') {
248 						p.lastIndex = 0;
249 					}
250 					if (!options.hasOwnProperty(name) || inputField===null || inputField===undefined || inputField==='') {
251 						error = {
252 							field : name,
253 							type : 'RequiredError',
254 							message : '{fieldnames.' + name + '} is required'
255 						};
256 					}
257 					else if (pType==='regex' && !p.test(inputField+'')) {
258 						error = {
259 							field : name,
260 							type : 'ValidationError',
261 							message : '{fieldnames.' + name + '} is invalid'
262 						};
263 					}
264 					else if (pType==='type' && p!==puredom.typeOf(inputField)) {
265 						error = {
266 							field : name,
267 							type : 'TypeError',
268 							message : '{fieldnames.' + name + '} is invalid'
269 						};
270 					}
271 					if (error) {
272 						error.missingParameter = error.field;
273 						response.message += (response.message.length>0?', ':'') + error.message + ' ('+error.type+')';
274 						response.errors.push(puredom.extend(error, baseErrorMessage));
275 					}
276 				}
277 			}
278 		}
279 		
280 		response.failed = response.errors.length>0;
281 		return response;
282 	};

286 	createApiMethod = function(subject, action, method) {
287 		self[subject][action] = function (options) {
288 			var validationResponse = priv.validateParameters(options=options||{}, method),
289 				callback, req, type, querystring, x, i,
290 				funcs, postData, requestParameters,
291 				isLongPolling = method.longPolling===true || (method.allowLongPolling===true && options.longPolling===true),
292 				optionsHasLongPollingProperty = options.hasOwnProperty('longPolling'),
293 				optionsLongPollingProperty = options.longPolling,
294 				unCache;
295 			
296 			// Check for a failed validation
297 			if (validationResponse.failed) {
298 				log("api."+subject+"."+action+": Errors: " + validationResponse.message, 9);
299 				
300 				// for historical reasons, return the first error on its own, then return the error collection as a third param:
301 				options.callback(false, validationResponse.errors[0], validationResponse.errors);
302 				return false;
303 			}
304 			
305 			// callback for JSONp (or other request methods)
306 			callback = function(json, extraData) {
307 				var success = json && (json[api.statusProperty || 'success']===true || json[api.statusProperty || 'success']===1),			// The response.success property MUST be Boolean TRUE or Integer 1  <---
308 					data = json.data || json || null,
309 					message = json.errorMessage || json.message || null,
310 					sval = json[api.statusProperty || 'success'],
311 					optionsLongPollingRef,
312 					enhancedMessage;
313 				
314 				// Some enhanced functionality used by error messages.
315 				extraData = extraData || {};
316 				extraData.apiMethod = method.endpoint;
317 				
318 				if (extraData.cached!==true && method.cache===true && self.enableCache===true) {
319 					priv.cacheResponse(method, options, json);
320 				}
321 				
322 				if (method.verifyResult) {
323 					success = method.verifyResult(json);
324 				}
325 				else if ((sval!==true && sval!==false && sval!==0 && sval!==1) || (json.constructor!==objConstructor && !isArray(json))) {
326 					success = (json.constructor===objConstructor || isArray(json)) ? true : false;
327 					data = json;
328 					message = null;
329 				}
330 				
331 				// Long Polling!
332 				if (isLongPolling) {
333 					// Did the server respond with "timedout":true?
334 					if (success && json.timedout===true) {
335 						// Looks like we need to re-initiate the request:
336 						optionsLongPollingRef = options;
337 						
338 						setTimeout(function() {
339 							self[subject][action](optionsLongPollingRef);
340 							callback = optionsLongPollingRef = success = json = data = message = sval = null;
341 						}, 1);
342 						// We'll respond later.
343 						return true;
344 					}
345 				}
346 				
347 				// Complex response objects allow for passing more data using the existing structure:
348 				data = createNativeAPIResponse(data, json, extraData);
349 				enhancedMessage = new MessageStringWithData(json, message);
350 				
351 				if (extraData.parseError===true) {
352 					success = false;
353 					data = data.message || data;
354 					if (api.onParseError) {
355 						api.onParseError(json, extraData);
356 					}
357 				}
358 				
359 				(method.onbeforecomplete || method.onBeforeComplete || method.precallback || emptyFunc)(success, success===true?data:enhancedMessage, json);
360 				if (success===true) {
361 					(options.onsuccess || method.onsuccess || emptyFunc).call(api.endpoints[subject][action], data, json);
362 				}
363 				else {
364 					(options.onerror || method.onerror || emptyFunc).call(api.endpoints[subject][action], message, data, json);
365 				}
366 				(options.oncomplete || options.callback || emptyFunc).call(api.endpoints[subject][action], success,success===true?data:enhancedMessage, json);
367 				(method.oncomplete || method.onComplete || method.callback || emptyFunc)(success, success===true?data:enhancedMessage, json);
368 				if (api.onRequestCompleted) {
369 					api.onRequestCompleted(subject+'.'+action, data, success, requestParameters, options);
370 				}
371 				callback = req = type = querystring = funcs = postData = requestParameters = isLongPolling = optionsHasLongPollingProperty = optionsLongPollingProperty = unCache = enhancedMessage = null;
372 				options = p = null;
373 			};//-callback
374 			
375 			// general request prep
376 			type = (method.type && method.type.toLowerCase()) || "";
377 			querystring = method.endpoint;
378 			if (method.formatSuffix) {
379 				querystring += method.formatSuffix;
380 			}
381 			else if (api.formatSuffix) {
382 				querystring += api.formatSuffix;
383 			}
384 			if (!querystring.match(/^(http|https|ftp)\:/)) {
385 				querystring = api.root + querystring;
386 			}
387 			
388 			unCache = options._cache===false || options._nocache===true;
389 			//delete options._cache;
390 			//delete options._nocache;
391 			
392 			if (isLongPolling) {
393 				options.longPolling = null;
394 				try{ delete options.longPolling; }catch(err){}
395 				options.timeout = options.timeout || method.longPollingTimeout || self.longPollingTimeout || 60;
396 			}
397 			else if (method.cache===true && self.enableCache===true) {
398 				//console.log(method.endpoint, puredom.extend({}, options));
399 				if (unCache) {
400 					priv.uncache(method, options);
401 					if (options._cache_deleteonly===true) {
402 						return;
403 					}
404 				}
405 				else {
406 					i = priv.getCached(method, options);
407 					if (i) {
408 						setTimeout(function() {
409 							callback(i, {
410 								cached : true,
411 								fresh : false
412 							});
413 						}, 1);
414 						return;
415 					}
416 				}
417 			}
418 			
419 			//console.log('NativeAPI::querystring = ' + querystring);
420 			if (options) {
421 				querystring = querystring.replace(/\{([a-z0-9\-\._]+)\}/gim, function(s, i) {
422 					//console.log('NativeAPI::tpl('+i+', '+((options.hasOwnProperty(i) && options[i]!==null && options[i]!==undefined)?'true':'false')+')');
423 					if (options.hasOwnProperty(i) && options[i]!==null && options[i]!==undefined) {
424 						try {
425 							delete options[i];
426 						} catch(err) {
427 							options[i] = null;
428 						}
429 						return options[i];
430 					}
431 					return s;
432 				});
433 			}
434 			
435 			
436 			// specific request prep
437 			switch (type) {
438 				case "xdr":
439 					log("Cross-domain requests are not yet supported.", 7);
440 					break;
441 				
442 				case "post":
443 					funcs = {};
444 					for (var j in options) {
445 						if (options.hasOwnProperty(j) && Object.prototype.toString.apply(options[j])==="[object Function]") {
446 							funcs[j] = options[j];
447 							options[j] = null;
448 							try{ delete options[j]; }catch(err2){}
449 						}
450 					}
451 					requestParameters = postData = shallowObjectCopy(
452 						{},
453 						globalParameters,
454 						method.auth===true ? authParameters : {},
455 						method.defaultParameters || {},
456 						options
457 					);
458 					for (j in funcs) {
459 						if (funcs.hasOwnProperty(j)) {
460 							options[j] = funcs[j];
461 						}
462 					}
463 					funcs = null;
464 					
465 					// make the POST request
466 					puredom.net.request({
467 						url : querystring,
468 						type : "POST",
469 						post : postData,
470 						callback : function(success, response) {
471 							if (success && response) {
472 								callback(response);
473 							}
474 							else {
475 								if (this.jsonParseError===true) {
476 									callback({status:false, message:"Unable to parse server response", rawdata:this.responseText}, {parseError:true, clientsideErrorDetection:true});
477 								}
478 								else {
479 									callback({status:false, message:"Connection error "+this.status}, {clientsideErrorDetection:true});
480 								}
481 							}
482 						},
483 						contentTypeOverride : 'application/json'
484 					});
485 					break;
486 				
487 				case "jsonp":
488 					// TODO: re-write the following line, it does not handle arrays properly:
489 					requestParameters = shallowObjectCopy(
490 						{},
491 						globalParameters,
492 						method.auth===true ? authParameters : {},
493 						method.defaultParameters || {},
494 						options
495 					);
496 					querystring += getQueryStringFromObj(requestParameters);
497 					
498 					// replace the first "&" with a "?"
499 					if (querystring.indexOf("?")===-1 || querystring.indexOf("?")>querystring.indexOf("&")) {
500 						querystring = querystring.replace("&","?");
501 					}
502 					
503 					//console.log('puredom.net.jsonp(', querystring, callback, ');');
504 					
505 					// make the JSONp call
506 					req = puredom.net.jsonp(querystring, callback);
507 					break;
508 				
509 				//case "get":
510 				default:
511 					// TODO: re-write the following line, it does not handle arrays properly:
512 					requestParameters = shallowObjectCopy(
513 						{},
514 						globalParameters,
515 						method.auth===true ? authParameters : {},
516 						method.defaultParameters || {},
517 						options
518 					);
519 					querystring += getQueryStringFromObj(requestParameters);
520 					
521 					// replace the first "&" with a "?"
522 					if (querystring.indexOf("?")===-1 || querystring.indexOf("?")>querystring.indexOf("&")) {
523 						querystring = querystring.replace("&","?");
524 					}
525 					
526 					// make the GET request
527 					puredom.net.request({
528 						url : querystring,
529 						type : "GET",
530 						callback : function(success, response) {
531 							if (success && response) {
532 								callback(response);
533 							}
534 							else {
535 								if (this.jsonParseError===true) {
536 									callback({status:false, message:"Unable to parse server response", rawdata:this.responseText}, {parseError:true, clientsideErrorDetection:true});
537 								}
538 								else {
539 									callback({status:false, message:"Connection error "+this.status}, {clientsideErrorDetection:true});
540 								}
541 							}
542 						},
543 						contentTypeOverride : 'application/json'
544 					});
545 					break;
546 			}
547 			
548 			if (optionsHasLongPollingProperty) {
549 				options.longPolling = optionsLongPollingProperty;
550 			}
551 			
552 			return req;
553 		};
554 	};

558 		self = (function(obj) {
559 			/**	@ignore */
560 			function NativeAPI(){}
561 			for (var i in obj) {
562 				if (obj.hasOwnProperty(i)) {
563 					NativeAPI.prototype[i] = obj[i];
564 				}
565 			}
566 			return new NativeAPI();
567 		}(self));
568 	}

571 	var subject, action;
572 	for (subject in api.endpoints) {
573 		if (api.endpoints.hasOwnProperty(subject)) {
574 			self[subject] = new NativeAPIMethod();
575 			for (action in api.endpoints[subject]) {
576 				if (api.endpoints[subject].hasOwnProperty(action)) {
577 					createApiMethod(subject, action, api.endpoints[subject][action]);
578 				}
579 			}
580 		}
581 	}
582 	
583 	if (self.globalParameters) {
584 		for (var o in self.globalParameters) {
585 			if (self.globalParameters.hasOwnProperty(o)) {
586 				self.setGlobalParameter(o, self.globalParameters[o]);
587 			}
588 		}
589 	}
590 	
591 	if (this.constructor!==arguments.callee) {
592 		return self;
593 	}
