1 /**	Provides a cross-browser persisted storage layer using various storage adapters.
  2  *	@constructor Asynchronously creates an instance of LocalStorage.<br />
  3  *	<strong>Available Options:</strong><br />
  4  *	<table class="options"><thead>
  5  *		<tr><td>Option</td><td>Type</td><td>Default Value</td><td>Description</td></tr>
  6  *	</thead><tbody>
  7  *		<tr><td>adapter</td><td>String</td><td><em>auto-detect</em></td><td>Attempt to use a specific adapter. If unset, the best adapter is automatically used (useBest=true).</td></tr>
  8  *		<tr><td>useBest</td><td>Boolean</td><td>true</td><td>Attempt to use the best adapter available, unless an adapter is manually specified and loads successfully.</td></tr>
  9  *		<tr><td>restore</td><td>Boolean</td><td>true</td><td>Attempt to restore the data immediately.</td></tr>
 10  *	</tbody></table>
 11  *	@param {String} id				Required identifier for the specific storage instance.
 12  *	@param {Object} [options]		Hashmap of available config options (see description)
 13  *	@param {Function} [callback]	Gets passed a reference to the instance after the initial restore() has completed.
 14  */
 15 puredom.LocalStorage = function LocalStorage(id, callback, options) {
 16 	var self = this;
 17 	if (typeof arguments[2]==='function') {
 18 		callback = arguments[2];
 19 		options = arguments[1];
 20 	}
 21 	options = options || {};
 22 	
 23 	this.id = id;
 24 	this.adapter = null;
 25 	this.data = {};
 26 	
 27 	if (options.adapter) {
 28 		this.setAdapter(options.adapter);
 29 	}
 30 	if (!this.adapter && options.useBest!==false) {
 31 		this.useBestAdapter();
 32 	}
 33 	if (this.adapter && options.restore!==false) {
 34 		this.restore(function() {
 35 			if (callback) {
 36 				callback(self);
 37 			}
 38 			self = options = null;
 39 		});
 40 	}
 41 	else if (callback) {
 42 		callback(self);
 43 		self = null;
 44 	}
 45 };

 60 /** Specify the name of a storage adapter the instance should use.<br />
 61  *	<strong>Note:</strong> Pay attention to the return value! 
 62  *	Even if you know a given adapter exists, it may fail to load if it is not supported in the 
 63  *	current environment (eg: if window.localStorage doesn't exists, cookies are blocked, etc).
 64  *	@param {String} type		The name of an adapter to use. For a list, see {@link puredom.LocalStorage.adapters}
 65  *	@returns {Boolean} <code>true</code> if the adapter loaded successfully, <code>false</code> if the specified adapter did not exist or could not be loaded.
 66  */
 67 puredom.LocalStorage.prototype.setAdapter = function(type) {
 68 	var list = this.constructor.adapters,
 69 		lcType = (type+'').toLowerCase(),
 70 		found = false,
 71 		foundWorking = false,
 72 		i;
 73 	for (i in list) {
 74 		if (list.hasOwnProperty(i) && (i+'').toLowerCase()===lcType) {
 75 			found = true;
 76 			if (list[i].test(this)===true) {
 77 				foundWorking = true;
 78 				this.adapterName = type;
 79 				this.adapter = list[i];
 80 				break;
 81 			}
 82 		}
 83 	}
 84 	if (!found) {
 85 		puredom.log('puredom.LocalStorage :: Could not find "'+type+'" adapter.');
 86 		return false;
 87 	}
 88 	if (!foundWorking) {
 89 		puredom.log('puredom.LocalStorage :: "'+type+'" adapter test() failed: conditions for adapter use not met.');
 90 		return false;
 91 	}
 92 	return true;
 93 };

102 /**	Load whichever adapter works best in the current environment. <br />
103  *	This is determined by querying each adapter to check which are supported, then 
104  *	selecting the best based on a pre-determined "score", as reported by the adapter.<br />
105  *	<strong>Note:</strong> This method throws a delayed error (does not stop execution) if no adapters are supported in the current environment.
106  */
107 puredom.LocalStorage.prototype.useBestAdapter = function() {
108 	var list = this.constructor.adapters,
109 		best, bestName, i;
110 	for (i in list) {
111 		if (list.hasOwnProperty(i) && i!=='none' && list[i].test(this)===true) {
112 			if (!best || (Math.round(best.rating) || 0)<(Math.round(list[i].rating) || 0)) {
113 				best = list[i];
114 				bestName = i;
115 			}
116 		}
117 	}
118 	if (best) {
119 		this.adapterName = bestName;
120 		this.adapter = best;
121 	}
122 	else {
123 		setTimeout(function() {
124 			throw('puredom.LocalStorage :: Could not find the best adapter.');
125 		}, 1);
126 		return false;
127 	}
128 	return true;
129 };

131 /** Get a namespaced facade of the LocalStorage interface.<br />
132  *	<strong>Tip:</strong> This is a nice way to reduce the number of commits triggered by a large application, 
133  *	because all namespaces derived from a single LocalStorage instance share the same commit queue.
134  *	@param {String} ns		A namespace (prefix) to use. Example: <code>"model.session"</code>
135  *	@returns {puredom.LocalStorage.NamespacedLocalStorage} An interface identical to {@link puredom.LocalStorage}.
136  */
137 puredom.LocalStorage.prototype.getNamespace = function(ns) {
138 	var self = this,
139 		iface;
140 	ns = ns + '';
141 	if (ns.substring(0,1)==='.') {
142 		ns = ns.substring(1);
143 	}
144 	if (ns.substring(ns.length-1)==='.') {
145 		ns = ns.substring(0, ns.length-1);
146 	}
147 	iface = puredom.extend(new puredom.LocalStorage.NamespacedLocalStorage(), {
148 		getAdapter : function() {
149 			return self.getAdapter();
150 		},
151 		/** omg this is so meta */
152 		getNamespace : this.getNamespace,
153 		getValue : function(key) {
154 			return self.getValue(ns+'.'+key);
155 		},
156 		setValue : function(key, value) {
157 			self.setValue(ns+'.'+key, value);
158 			return this;
159 		},
160 		removeKey : function(key) {
161 			self.removeKey(ns+'.'+key);
162 			return this;
163 		},
164 		purge : function() {
165 			self.removeKey(ns);
166 			return this;
167 		},
168 		getData : function() {
169 			return self.getValue(ns);
170 		},
171 		restore : function(callback) {
172 			var proxiedCallback,
173 				proxiedContext = this;
174 			if (callback) {
175 				proxiedCallback = function() {
176 					callback(proxiedContext);
177 					proxiedContext = proxiedCallback = callback = null;
178 				};
179 			}
180 			self.restore(proxiedCallback);
181 			return this;
182 		},
183 		commit : function(callback) {
184 			var proxiedCallback,
185 				proxiedContext = this;
186 			if (callback) {
187 				proxiedCallback = function() {
188 					callback(proxiedContext);
189 					proxiedContext = proxiedCallback = callback = null;
190 				};
191 			}
192 			self.commit(proxiedCallback);
193 			return this;
194 		}
195 	});
196 	puredom.extend(iface, {
197 		get : iface.getValue,
198 		set : iface.setValue,
199 		remove : iface.removeKey
200 	});
201 	return iface;
202 };

212 /** Get the stored value corresponding to a dot-notated key.
213  *	@param {String} key		A key, specified in dot-notation.
214  *	@returns If <code>key</code> exists, returns the corresponding value, otherwise returns undefined.
215  */
216 puredom.LocalStorage.prototype.getValue = function(key) {
217 	var value = puredom.delve(this.data, key);
218 	return value;
219 };
220 
221 /** Set the stored value corresponding to a dot-notated key. <br />
222  *	If the key does not exist, it is created.
223  *	@param {String} key		A key, specified in dot-notation.
224  *	@param {Any} [value]	The value to set. If an {Object} or {Array}, its internal values become accessible as dot-notated keys. If <code>null</code> or <code>undefined</code>, the key is removed.
225  *	@returns {this}
226  */
227 puredom.LocalStorage.prototype.setValue = function(key, value) {
228 	var node = this.data,
229 		keyParts = key.split('.'),
230 		i;
231 	for (i=0; i<keyParts.length-1; i++) {
232 		if (!node.hasOwnProperty(keyParts[i])) {
233 			node[keyParts[i]] = {};
234 		}
235 		node = node[keyParts[i]];
236 	}
237 	if (value===undefined || value===null) {
238 		node[keyParts[keyParts.length-1]] = null;
239 		delete node[keyParts[keyParts.length-1]];
240 	}
241 	else {
242 		node[keyParts[keyParts.length-1]] = value;
243 	}
244 	this.queueCommit();
245 	return this;
246 };

248 /** Remove a key and (its stored value) from the collection.
249  *	@param {String} key		A key, specified in dot-notation.
250  *	@returns {this}
251  */
252 puredom.LocalStorage.prototype.removeKey = function(key) {
253 	this.setValue(key, undefined);
254 	return this;
255 };

275 /** Restore the collection from its persisted state.
276  *	@param {Function} callback		Gets called when the restore has completed, passed a reference to the instance.
277  *	@returns {this}
278  */
279 puredom.LocalStorage.prototype.restore = function(callback) {
280 	var self = this,
281 		data, asyncData;
282 	data = this._adapterCall('load', function(r) {
283 		self.data = asyncData = r || {};
284 		if (callback) {
285 			callback(self);
286 		}
287 		self = null;
288 	});
289 	if (data && !asyncData) {
290 		this.data = data;
291 		if (callback) {
292 			callback(this);
293 		}
294 	}
295 	data = asyncData = null;
296 	return this;
297 };

299 /** Save the collection <strong>immediately</strong> using the active persistence adapter.<br />
300  *	This bypasses the default "delayed write" save technique that is implicitly used when interacting with other methods.
301  *	@param {Function} callback		Gets called when the commit has completed, passed a reference to the instance.
302  *	@returns {this}
303  */
304 puredom.LocalStorage.prototype.commit = function(callback) {
305 	var self = this;
306 	if (this._commitTimer) {
307 		clearTimeout(this._commitTimer);
308 		this._commitTimer = null;
309 	}
310 	this._adapterCall('save', this.data, function() {
311 		if (callback) {
312 			callback(self);
313 		}
314 		self = null;
315 	});
316 	return this;
317 };
318 
319 /** Queue a commit if one is not already queued.
320  *	@private
321  */
322 puredom.LocalStorage.prototype.queueCommit = function() {
323 	var self = this;
324 	if (!this._commitTimer) {
325 		this._commitTimer = setTimeout(function() {
326 			self.commit();
327 			self = null;
328 		}, this.commitDelay);
329 	}
330 };
331 
332 /** Make a call to the active persistence adapter.
333  *	@private
334  *	@param {String} func	The adapter function to execute
335  *	@param args				All other arguments are forwarded on to the adapter.
336  *	@returns {Any} The adapter method's return value.
337  */
338 puredom.LocalStorage.prototype._adapterCall = function(func, args) {
339 	if (this.adapter && this.adapter[func]) {
340 		return this.adapter[func].apply(this.adapter, [this].concat(puredom.toArray(arguments).slice(1)));
341 	}
342 };

362 /** Register a storage adapter.
363  *	@param {String} name		A name for the adapter. Used by {@link puredom.LocalStorage#setAdapter} and {@link puredom.LocalStorage#getAdapter}.
364  *	@param {Object} adapter		The adapter itself.
365  *	@public
366  */
367 puredom.LocalStorage.addAdapter = function(name, adapter) {
368 	if (!adapter.save) {
369 		throw('puredom.LocalStorage :: Adapter "'+name+'" attempted to register, but does not provide a save() method.');
370 	}
371 	else  if (!adapter.load) {
372 		throw('puredom.LocalStorage :: Adapter "'+name+'" attempted to register, but does not provide a load() method.');
373 	}
374 	else {
375 		this.adapters[name] = adapter;
376 	}
377 };

383 puredom.extend(puredom.LocalStorage.adapters.none.prototype, /** @lends puredom.LocalStorage.adapters.none */ {
384 	
385 	/** The default ID to use for database storage. <br />
386 	 *	This is used as a fallback in cases {@link puredom.LocalStorage#id} does not exist.
387 	 */
388 	defaultName : 'db',
389 	
390 	/** An adapter rating from 0-100. Ratings should be based on <strong>speed</strong> and <strong>storage capacity</strong>. <br />
391 	 *	It is also possible to produce a dynamic rating value based on the current environment, though this is not recommended in most cases.
392 	 */
393 	rating : 0,
394 	
395 	/** Tests if the adapter is supported in the current environment.
396 	 *	@param {puredom.LocalStorage} storage		The parent storage instance.
397 	 *	@returns {Boolean} isSupported
398 	 */
399 	test : function(storage) {
400 		return false;
401 	},
402 	
403 	/** Load all persisted data.
404 	 *	@param {puredom.LocalStorage} storage	The parent storage instance.
405 	 *	@param {Function} callback				A function to call once the data has been loaded. Expects a JSON object.
406 	 */
407 	load : function(storage, callback) {
408 		if (callback) {
409 			callback();
410 		}
411 	},
412 	
413 	/** Save all data to the persistence layer.
414 	 *	@param {puredom.LocalStorage} storage	The parent storage instance.
415 	 *	@param {Object} data					The JSON data to be saved.
416 	 *	@param {Function} callback				A function to call once the data has been saved. Expects a {Boolean} value indicating if the save was successful.
417 	 */
418 	save : function(storage, data, callback) {
419 		if (callback) {
420 			callback(false);
421 		}
422 	}
423 	
424 });
