1 /**	Fire events and listen for fired events. <br />
  2  *	Let's just assume every framework provides one of these now.
  3  *	@constructor Creates a new EventEmitter instance.
  4  */
  5 puredom.EventEmitter = function EventEmitter() {
  6 	this._eventRegistry = [];
  7 };
  8 
  9 
 10 /** Register an event listener on the instance.
 11  *	@param {String} type		An event type, or a comma-seprated list of event types.
 12  *	@param {Function} handler	A function to call in response to events of the given type.
 13  *	@returns {this}
 14  */
 15 puredom.EventEmitter.prototype.addEventListener = function(type, handler) {
 16 	var t, i;
 17 	type = (type + '').toLowerCase().replace(/\s+/gim,'');
 18 	if (type.indexOf(',')>-1) {
 19 		t = type.split(',');
 20 		for (i=0; i<t.length; i++) {
 21 			this.addEventListener(t[i],handler);
 22 		}
 23 		return this;
 24 	}
 25 	type = type.replace(/^on/,'');
 26 	this._eventRegistry.push({
 27 		type : type,
 28 		handler : handler
 29 	});
 30 	return this;
 31 };
 32 
 33 /**	Alias of {@link puredom.EventEmitter#addEventListener}
 34  *	@function
 35  *	@private
 36  */
 37 puredom.EventEmitter.prototype.addListener = puredom.EventEmitter.prototype.addEventListener;
 38 
 39 /**	Alias of {@link puredom.EventEmitter#addEventListener}
 40  *	@function
 41  */
 42 puredom.EventEmitter.prototype.on = puredom.EventEmitter.prototype.addEventListener;

 44 
 45 /** Remove an event listener from the instance.
 46  *	@param {String} type		An event type, or a comma-seprated list of event types.
 47  *	@param {Function} handler	A reference to the handler, as was originally passed to {puredom.EventEmitter#addEventListener}.
 48  *	@returns {this}
 49  */
 50 puredom.EventEmitter.prototype.removeEventListener = function(type, handler) {
 51 	var x, r, t, i;
 52 	type = (type + '').toLowerCase().replace(/\s+/gim,'');
 53 	if (type.indexOf(',')>-1) {
 54 		t = type.split(',');
 55 		for (i=0; i<t.length; i++) {
 56 			this.removeEventListener(t[i],handler);
 57 		}
 58 		return this;
 59 	}
 60 	type = type.replace(/^on/,'');
 61 	for (x=this._eventRegistry.length; x--; ) {
 62 		r = this._eventRegistry[x];
 63 		if (r.type===type && r.handler===handler) {
 64 			this._eventRegistry.splice(x, 1);
 65 			break;
 66 		}
 67 	}
 68 	return this;
 69 };
 70 
 71 /**	Alias of {@link puredom.EventEmitter#removeEventListener}
 72  *	@function
 73  *	@private
 74  */
 75 puredom.EventEmitter.prototype.removeListener = puredom.EventEmitter.prototype.removeEventListener;
 76 
 77 
 78 /** Fire an event of a given type. <br />
 79  *	Pass a comma-separated list for <code>type</code> to fire multiple events at once.
 80  *	@param {String} type	An event type, or a comma-seprated list of event types.
 81  *	@param {Array} args		An Array of arguments to pass to each handler. Non-Array values get auto-boxed into an Array.
 82  *	@returns {Array} an Array of handler return values. The Array also has "truthy" and "falsey" properties indicating if any handlers returned <code>true</code> or <code>false</code>, respectively.
 83  */
 84 puredom.EventEmitter.prototype.fireEvent = function(type, args) {
 85 	var x, r, errors=[], rval, returns=[];
 86 	type = (type+'').toLowerCase().replace(/^on/,'');
 87 	if (!puredom.isArray(args)) {
 88 		args = [args];
 89 	}
 90 	for (x=this._eventRegistry.length; x--; ) {
 91 		r = this._eventRegistry[x];
 92 		if (r.type===type) {
 93 			if (returns.length===0) {
 94 				returns.falsy = returns.falsey = returns.truthy = true;
 95 			}
 96 			rval = r.handler.apply(this, args);
 97 			returns.push(rval);
 98 			if (rval===true) {
 99 				returns.falsy = returns.falsey = false;
100 			}
101 			else if (rval===false) {
102 				returns.truthy = false;
103 			}
104 			if (rval===false) {
105 				break;
106 			}
107 		}
108 	}
109 	errors = null;
110 	return returns;
111 };

113 /**	Alias of {@link puredom.EventEmitter#fireEvent}
114  *	@deprecated
115  *	@private
116  */
117 puredom.EventEmitter.prototype._fireEvent = puredom.EventEmitter.prototype.fireEvent;
